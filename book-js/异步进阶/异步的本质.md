
- 再次强调! JS单线程, 异步是跑不掉的, event-loop也跑不掉~

- ***async-await 只是一个语法糖, 但是这颗糖, 真香!***

```js
  async function foo() {
    console.log(1)  // 2
    await foo2()    // foo2()会立即执行, 然后再进行 await 操作, await行的后面都可以看做 callback 的内容, 即异步
    console.log(3)  // 5
    console.log(4)  // 6
  }

  async function foo2() {
    console.log(2)  // 3
  }

  console.log('start')  // 1
  foo()
  console.log('end')  // 4
```

```js
  async function foo() {
    console.log(1)  // 2
    await foo2()

    // callback
    console.log(3)  // 5
    await foo3()

    // callback
    console.log(4)  // 7
  }

  async function foo2() {
    console.log(2)  // 3
  }

  async function foo3() {
    console.log(5)  // 6
  }

  console.log('start')  // 1
  foo()
  console.log('end')  // 4
```
